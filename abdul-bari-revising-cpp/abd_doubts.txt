#include <iostream>
using namespace std;
class complex
{
    friend void operator<<(ostream&, complex&);
    int real;
    int img;

public:
    complex(int r = 0, int i = 0) : real(r), img(i) {}
};

ostream &operator<<(ostream &o, complex &c)
{
    o << c.real<< " + i"<<c.img<<endl;
    return o;
}
int main()
{
    complex c1(3, 7);
    cout<<c1;
}
